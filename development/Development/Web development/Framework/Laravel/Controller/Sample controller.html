<?php

//Basic declaration 
namespace App\Http\Controllers\API;
use App\Http\Controllers\Controller;
use Illuminate\Http\Request;

//Declare Model 
use App\Forum;

class forumController extends Controller
{
    //Controller function 
    public function getForum()
    {

        //To get all the data from table Forum 
        return Forum::all();

    }
 
    public function searchForum($id)
    {

        //To find id from the table Forum primary key
$id = Forum::find($id);
        return $id;

    }

    public function createForum(Request $request)
    {

        //To create new table Forum data 
        $forum= Forum::create($request->all());

        //Return JSON response 
        return response()->json($forum, 201);

    }

    public function updateForum(Request $request,$id)
    {

        //To update table Forum data
        $forum = Forum::find($id);
        $forum->update($request->all());
        return response()->json($forum,200);
    }

    public function deleteForum($id)
    {

        //To find and delete specific table Forum data and return message in JSON type
$forum = Forum::find($id);
        $forum->delete();
        return response()->json(['success'=>true,'desc'=>$id->content.'deleted']);
    }
}
