Step 2 : Setup API route ( routes/api.php ) 

  Route::post('add-image', 'API\imageController@ajaxImageUploadPost');



Step 3 : Create controller for the route ( app/Http/Controllers/API/imageController )

 
 public function ajaxImageUploadPost(Request $request)
 {
$imageName = $request->id . '.' . 
$request->file('image')->getClientOriginalExtension();

$request->file('image')->move(
base_path() . '/public/images/catalog/', $imageName
);

return \Redirect::route('admin.products.edit', 
array($request->id))->with('message', 'Product added!');  
   }
